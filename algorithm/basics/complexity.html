# 为什么需要复杂度分析

我们都知道，数据结构和算法本身解决的是“快”和“省”的问题，那如何来衡量这个“快”和“省”了？

所以，我们需要复杂度分析。复杂度分析包括：时间和空间 复杂度。

介绍下不用具体的测试数据来测试，就可以粗略地估计算法的执行效率的方法，大O复杂度表示法。

# 大 O 复杂度表示法

先看一个问题，求1，2，3...n的累加和，代码如下：

// 因为是错略估计，所以假设每行代码执行时间一样 === unit_time，简写 t 。
function cal(n) {
    let sum = 0;                // 执行时间 1t
    let i = 1;                  // 执行时间 1t
    for (; i <= n; ++i) {       // 执行时间 nt
        sum = sum + i;          // 执行时间 nt
    }                           // 所以，执行总时间 T(n) = (2n+2)t
    return sum;                 
}

通过上面代码分析得出，所有代码的执行时间 T(n) 与每行代码的总执行次数(2n+2)成正比

再看下面的代码：

function cal(n) { 
    let sum = 0;                   // 1t
    let i = 1;                     // 1t
    let j = 1;                     // 1t
    for (; i <= n; ++i) {          // 循环n次，n*t
        j = 1;                     // 循环n次，n*t
        for (; j <= n; ++j) {      // 嵌套的循环里面，(n*n)*t
            sum = sum + i * j;     // 嵌套的循环里面，(n*n)*t
        }                          // 所以，执行总时间 T(n) = (2n*n+2n+3)t
    } 
}

同样可以得出，所有代码的执行时间 T(n) 与每行代码的总执行次数(2n*n+2n+3)成正比

把这种规律总结成公式，

  • T(n) 表示代码执行的时间
  • n 表示数据规模的大小
  • f(n) 表示每行代码执行的次数总和，这里f(n) 表示上面的(2n+2)、(2n*n+2n+3)
  • 公式中的 O，表示代码的执行时间 T(n) 与 f(n) 表达式成正比

这就是大 O 时间复杂度表示法。

大 O 时间复杂度实际上并不具体表示代码真正的执行时间，而是表示代码执行时间随数据规模增长的变化趋势，所以，也叫作渐进时间复杂度（asymptotic time complexity），简称时间复杂度。

当 n 很大时，比如10000、100000，可以忽略公式中的常量和系数，我们只需要记录一个最大量级就可以了，如果用大 O 表示法表示刚讲的那两段代码的时间复杂度，就可以记为：T(n) = O(n)和T(n) = O(n2)

# 时间复杂度分析

具体分析一段代码的复杂度，推荐三种方式：

# 只关注循环执行次数最多的一段代码

大 O 复杂度表示方法只是表示一种变化趋势，通常会忽略掉公式中的常量、低阶、系数，只需要记录一个最大阶的量级。所以，我们在分析一段代码的时间复杂度的时候，也只关注循环执行次数最多的那一段代码就可以了。

function cal(n) {
    let sum = 0;                
    let i = 1;                  
    for (; i <= n; ++i) {     // 循环部分是最大量级
        sum = sum + i;          
    }                          
    return sum;                 
}

# 加法法则

加法法则：总复杂度等于量级最大的那段代码的复杂度

function cal(n) { 
    let sum_1 = 0; 
    let p = 1; 
    for (; p < 100; ++p) {   // 循环100，常量,O(1)
        sum_1 = sum_1 + p; 
    } 
    
    let sum_2 = 0; 
    let q = 1; 

    for (; q < n; ++q) {    // 循环n，O(n)
        sum_2 = sum_2 + q; 
    } 
    
    let sum_3 = 0; 
    let i = 1; 
    let j = 1; 

    for (; i <= n; ++i) {     // 循环n*n,n的平方，O(n2)
        j = 1; 
        for (; j <= n; ++j) { 
            sum_3 = sum_3 + i * j; 
        } 
    } 
    
    return sum_1 + sum_2 + sum_3; 
}

# 乘法法则

乘法法则：嵌套代码的复杂度等于嵌套内外代码复杂度的乘积

function f(n) { 
    let sum = 0; 
    let i = 1; 
    for (; i < n; ++i) {   // 循环n，复杂度 O(n)
        sum = sum + i; 
    } 
    return sum; 
}
function cal(n) { 
    let ret = 0; let i = 1; 
    for (; i < n; ++i) {    // 循环n，复杂度O(n)
        ret = ret + f(i);   // 循环n，调用f(i),复杂度O(n)*O(n),等于O(n2)
    } 
}

# 几种常见时间复杂度实例分析

常见复杂度量级：

  1. O(1)

O(1) 是常量级时间复杂度的一种表示方法。总之可以明确知道代码执行次数的都是O(1)。

一般情况下，只要算法中不存在循环语句、递归语句,即使有成千上万行的代码，其时间复杂度也是常量级，即Ο(1)。

  1. O(logn)、O(nlogn)

对数阶时间复杂度非常常见，同时也是最难分析的一种时间复杂度。如下代码：

function cal(n){
    let i=1; 
    while (i <= n) { 
        i = i * 2; 
    }
}

根据我们前面讲的复杂度分析方法，第三、四行代码是循环执行次数最多的。所以，我们只要能计算出这部分循环代码被执行了多少次，就能知道整段代码的时间复杂度。

从代码中可以看出，变量 i 的值从 1 开始取，每循环一次就乘以 2。当大于 n 时，循环结束。 实际上，变量 i 的取值就是一个等比数列。如果我把它一个一个列出来，就应该是这个样子的：

所以，我们只要知道 x 值是多少，就知道这行代码执行的次数了。通过 2的x方=n 求解 x，就是求x的对数，即x=log2n，所以，这段代码的时间复杂度就是 O(log2n)。

对数复杂度统一记为：O(logn)

因为对数之间是可以互相转换的，log3n 就等于 log32 * log2n，所以 O(log3n) = O(C * log2n)，其中 C=log32 是一个常量。基于我们前面的一个理论：在采用大 O 标记复杂度的时候，可以忽略系数，即 O(Cf(n)) = O(f(n))。所以，O(log2n) 就等于 O(log3n)。因此，在对数阶时间复杂度的表示方法里，我们忽略对数的“底”， 统一表示为 O(logn)。

  1. O(m+n)、O(m*n)

这种情况代码的复杂度由两个数据的规模来决定，如下代码：

function cal(m, n) { 
    let sum_1 = 0; 
    let i = 1; 
    for (; i < m; ++i) { 
        sum_1 = sum_1 + i; 
    } 
    
    let sum_2 = 0; 
    let j = 1; 
    
    for (; j < n; ++j) { 
        sum_2 = sum_2 + j; 
    } 
    
    return sum_1 + sum_2;
}

# 空间复杂度分析

类比时间复杂度，空间复杂度全称就是渐进空间复杂度（asymptotic space complexity），表示算法的存储空间与数据规模之间的增长关系。

// 为了说明空间复杂性，不讨论代码是否合理
function test(n){
    let i = 0；             // 申请了空间，是个常量，所以空间复杂度O(1)
    let arr = [];           // 数组需要分配空间，循环里面可以看到数组长度是n-1，所以O(n)
    for(i;i<n;i++){         // 所以空间复杂度是O(n)
        arr[i] = i;
    }
    console.log(arr);
}

# 总结

复杂度也叫渐进复杂度，包括时间复杂度和空间复杂度，用来分析算法执行效率与数据规模之间的增长关系，可以粗略地表示，越高阶复杂度的算法，执行效率越低。
