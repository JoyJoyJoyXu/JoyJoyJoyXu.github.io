# New

new操作符做了哪些事：

  1. 创建一个新的对象
  2. 执行原型链[[prototype]]链接
  3. 将新创建的对象绑定到函数调用的this上
  4. 如果函数没有返回，就返回这个新的对象。
function New(func){
    var res = {};  // 创建新对象

    if(typeof func != 'function'){  // 判断是否函数
        throw TypeError('需要函数')
    }

    if(func.prototype != null){  // 绑定原型链
        res.setPrototypeOf(res, func.prototype)
    }

    var ret = func.apply(res, Array.prototype.slice.call(arguments,1))  //绑定函数调用的this

    if(ret != null && {typeof ret === 'object' || typeof ret === 'function'}){
        return ret;
    }

    return res;
}

# instanceOf

instanceOf 运算符用来判断一个构造函数的prototype是否在一个检测对象的原型链上

function instanceOf(left, right){
    let proto = Object.getPrototypeOf(left);
    let _proto = right.prototype;

    while(true){
        if(proto === null) return false;

        if(proto === _proto) return true;  

        proto = Object.getPrototypeOf(proto);
    }
}

# 防抖

防抖：如果短时间内大量触发同一事件，只会执行一次函数。

function debounce(func, delay){
    let timer = null;
    
    return function(){
        if(timer) clearTimeout(timer);

        timer = setTimeout(() => {
            func.apply(this,arguments)
        }, delay) 
    }
}

# 节流

节流：如果短时间内大量触发同一事件，那么在函数执行一次之后，该函数在指定的时间期限内不再工作，直至过了这段时间才重新生效。

function throttle(func, delay){
    let valid = true;

    return function(){
        if(!true){
            return false;
        }

        valid = false;

        settimeOut(() => {
            func.apply(this, arguments);
            valid = true;
        }, delay)
    }
}

# 日期转换

Date.prototype.format = function (fmt) {
    var o = {
        "M+": this.getMonth() + 1,                 //月份 
        "d+": this.getDate(),                    //日 
        "h+": this.getHours(),                   //小时 
        "m+": this.getMinutes(),                 //分 
        "s+": this.getSeconds(),                 //秒 
        "q+": Math.floor((this.getMonth() + 3) / 3), //季度 
        "S": this.getMilliseconds()             //毫秒 
    };
    if (/(y+)/.test(fmt)) {
        fmt = fmt.replace(RegExp.$1, (this.getFullYear() + "").substr(4 - RegExp.$1.length));
    }
    for (var k in o) {
        if (new RegExp("(" + k + ")").test(fmt)) {
            fmt = fmt.replace(RegExp.$1, (RegExp.$1.length == 1) ? (o[k]) : (("00" + o[k]).substr(("" + o[k]).length)));
        }
    }
    return fmt;
}

浅拷贝

Object.assign(target,obj1,obj2...)

target:目标

obj1,obj2...: 被复制对象

用ES6展开运算符，也可以。 let obj = {...obj1}

深拷贝

乞丐版：复制对象必须可以转换成JSON.

var newObj = JSON.parse( JSON.stringify( someObj ) );

简单递归：

function deepCopy(obj){
    if(typeof obj === 'object'){
        let result = obj.constructor == Array ? [] : {};
        for(let i in obj){
            result[i] = typeof obj[i] === 'object'? deepCopy(obj[i]):obj[i];
        }
    }else{
        // null ...
        return obj;
    }
}
